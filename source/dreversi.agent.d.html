24     Action select(in Board board, size_t depth = 3) {
25         Action best;
26         foreach (r; 0 .. board.length!0) {
27             foreach (c; 0 .. board.length!1) {
28                 auto next = board.put(this.isBlack, r, c);
29                 if (next.valid) {
30                     immutable score = search(next, false, depth);
31                     if (this.verbose) writefln!"Action(%d, %d, %f)"(r, c, score);
32                     if (best.score.isNaN || score > best.score) {
33                         best = Action(r, c, score);
34                     }
35                 }
36             }
37         }
38         return best;
39     }

41     Score search(in Board board, bool isMine, // true if board is result of my action
42                  size_t depth
43         ) {
44         import std.algorithm : max, min;
45 
46         if (board.finished || depth == 0) {
47             return cast(double) board.score(this.isBlack);
48         }
49 
50         /// pass
51         if ((isMine && board.pass(this.isBlack)) || (!isMine && board.pass(!this.isBlack))) {
52             return search(board, !isMine, depth-1);
53         }
54 
55         Score alpha = -double.infinity;
56         Score beta = double.infinity;
57         foreach (r; 0 .. board.length!0) {
58             foreach (c; 0 .. board.length!1) {
59                 immutable color = isMine ? this.isBlack : !this.isBlack; // XOR
60                 immutable b = board.put(color, r, c);
61                 if (b.valid) {
62                     const score = search(b, !isMine, depth - 1); // recur
63                     if (isMine) {
64                         alpha = max(alpha, score);
65                     } else {
66                         beta = min(beta, score);
67                     }
68                 }
69             }
70         }
71         return isMine ? alpha : beta;
72     }
73 }

142     Score search(bool isMine = false)(in Board board, size_t depth,
143                                       Score alpha = -double.infinity,
144                                       Score beta = double.infinity) {
145         // import std.algorithm : max, min;
146         import mir.math.common : fmax, fmin;
147 
148         static if (isMine) {
149             immutable scale = 1;
150         } else {
151             immutable scale = -1;
152         }
153 
154         if (board in this.scoreCache) {
155             return scale * this.scoreCache[board];
156         }
157 
158         if (board.finished || depth == 0) {
159             immutable s = cast(Score) board.score(this.isBlack);
160             this.memoizeScore(board, s);
161             return s;
162         }
163 
164         /// pass
165         static if (isMine) {
166             immutable color = this.isBlack;
167         } else {
168             immutable color = !this.isBlack;
169         }
170         if (board.pass(color)) {
171             immutable s = search!(!isMine)(board, depth-1, alpha, beta);
172             this.memoizeScore(board, s);
173             return s;
174         }
175 
176     outer: foreach (r; 0 .. board.length!0) {
177             foreach (c; 0 .. board.length!1) {
178                 immutable b = board.put(color, r, c);
179                 if (b.valid) {
180                     immutable score = search!(!isMine)(b, depth - 1, alpha, beta); // recur
181                     static if (isMine) {
182                         alpha = fmax(alpha, score);
183                     } else {
184                         beta = fmin(beta, score);
185                     }
186                     if (alpha >= beta) { // cut
187                         break outer;
188                     }
189                 }
190             }
191         }
192         static if (isMine) {
193             this.memoizeScore(board, alpha);
194             return alpha;
195         } else {
196             this.memoizeScore(board, -beta);
197             return beta;
198         }
199     }
