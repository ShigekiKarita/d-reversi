154 /// returns next step by the action
155 auto put(in Board b, bool isBlack, size_t row, size_t col) {
156     import std.algorithm : min;
157 
158     immutable p = isBlack ? Point.black : Point.white;
159     if (row >= b.length!0 || col >= b.length!1 || b[row, col] != Point.empty) {
160         return Board(b.data, false);
161     }
162 
163     auto ret = b.data.slice; // dup
164     // upper
165     foreach_reverse (r; 0 .. row) {
166         if (ret[r, col] == Point.empty) {
167             break;
168         }
169         if (ret[r, col] == p) {
170             ret[r..row, col] = p;
171             break;
172         }
173     }
174     // lower
175     foreach (r; row+1 .. ret.length!0) {
176         if (ret[r, col] == Point.empty) {
177             break;
178         }
179         if (ret[r, col] == p) {
180             ret[row+1..r, col] = p;
181             break;
182         }
183     }
184     // left
185     foreach_reverse (c; 0 .. col) {
186         if (ret[row, c] == Point.empty) {
187             break;
188         }
189         if (ret[row, c] == p) {
190             ret[row, c..col] = p;
191             break;
192         }
193     }
194     // right
195     foreach (c; col+1..ret.length!1) {
196         if (ret[row, c] == Point.empty) {
197             break;
198         }
199         if (ret[row, c] == p) {
200             ret[row, col+1..c] = p;
201             break;
202         }
203     }
204     // left-upper
205     foreach (n; 1..min(row, col) + 1) {
206         if (ret[row-n, col-n] == Point.empty) {
207             break;
208         }
209         if (ret[row-n, col-n] == p) {
210             foreach (m; 1 .. n+1) {
211                 ret[row-m, col-m] = p;
212             }
213             break;
214         }
215     }
216     // left-lower
217     foreach (n; 1..min(row, ret.length!1 - col - 1) + 1) {
218         if (ret[row-n, col+n] == Point.empty) {
219             break;
220         }
221         if (ret[row-n, col+n] == p) {
222             foreach (m; 1 .. n+1) {
223                 ret[row-m, col+m] = p;
224             }
225             break;
226         }
227     }
228     // right-upper
229     foreach (n; 1..min(ret.length!0 - row - 1, col) + 1) {
230         if (ret[row+n, col-n] == Point.empty) {
231             break;
232         }
233         if (ret[row+n, col-n] == p) {
234             foreach (m; 1 .. n+1) {
235                 ret[row+m, col-m] = p;
236             }
237             break;
238         }
239     }
240     // right-lower
241     foreach (n; 1..min(ret.length!0 - row - 1, ret.length!1 - col - 1) + 1) {
242         if (ret[row+n, col+n] == Point.empty) {
243             break;
244         }
245         if (ret[row+n, col+n] == p) {
246             foreach (m; 1 .. n+1) {
247                 ret[row+m, col+m] = p;
248             }
249             break;
250         }
251     }
252     // if nothing is changed, the action is invalid
253     immutable valid = ret != b.data;
254     ret[row, col] = p;
255     return Board(ret, valid);

265 /// returns true if no next move for the color
266 auto pass(in Board b, bool isBlack) {
267     import std.algorithm : any, map;
268     auto stat = b.count;
269     if (stat.byValue.map!"a == 0".any) return true;
270     foreach (i; 0 .. b.length!0) {
271         foreach (j; 0 .. b.length!1) {
272             if (b.put(isBlack, i, j).valid) {
273                 return false;
274             }
275         }
276     }
277     return true;
278 }

371 /// returns initial board
372 auto reset(size_t rows = 8, size_t cols = 8) {
373     auto s = uninitSlice!Point(rows, cols);
374     s[] = Point.empty;
375     s[rows / 2, cols / 2] = Point.black;
376     s[rows / 2, cols / 2 - 1] = Point.white;
377     s[rows / 2 - 1, cols / 2] = Point.white;
378     s[rows / 2 - 1, cols / 2 - 1] = Point.black;
379     return Board(s);
380 }
